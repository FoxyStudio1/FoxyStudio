function exportIndexHtml() {
  const id = getBinId();
  if (!id) { toast('Sem Bin ID!', 'err'); return; }

  const btn = event.target;
  btn.textContent = 'A gerar...'; btn.disabled = true;

  // Substituir o placeholder pelo BIN ID real
  const src = INDEX_HTML_TEMPLATE.replace("const BIN_ID = '%%BIN_ID%%';", "const BIN_ID = '" + id + "';");

  // Criar e descarregar ficheiro
  const blob = new Blob([src], { type: 'text/html' });
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  a.href = url; a.download = 'index.html';
  document.body.appendChild(a); a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  toast('index.html descarregado! Faz upload para o GitHub.', 'ok');
  btn.textContent = '⬇ Descarregar index.html'; btn.disabled = false;
}

async function saveAll() {
  setSaveIndicator('saving');
  collectSettings();
  const ok = await pushRemote(appData);
  setSaveIndicator(ok ? 'saved' : 'error');
  toast(ok ? 'Guardado com sucesso!' : 'Erro ao guardar!', ok ? 'ok' : 'err');
  setTimeout(() => setSaveIndicator('idle'), 3000);
}

function setSaveIndicator(state) {
  const el = document.getElementById('save-indicator');
  el.className = 'save-indicator';
  if (state === 'saving') { el.classList.add('saving'); el.textContent = 'A guardar...'; }
  else if (state === 'saved') { el.classList.add('saved'); el.textContent = '✓ Guardado'; }
  else if (state === 'error') { el.textContent = '✕ Erro'; el.style.color='#cc2200'; }
  else { el.textContent = '—'; }
}

async function init() {
  showBinStatus(getBinId());
  const remote = await fetchRemote();
  if (remote) {
    appData = { ...appData, ...remote };
    // ensure arrays exist
    if (!appData.projetos) appData.projetos = [];
    if (!appData.extraButtons) appData.extraButtons = [];
    if (!appData.settings) appData.settings = { backgroundType:'color', backgroundValue:'#000000' };
  }
  refreshPostUI();
  refreshProjetosUI();
  refreshExtraButtonsUI();
  loadSettingsUI();
}

function switchTab(name) {
  document.querySelectorAll('.tab-pane').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('tab-' + name).classList.add('active');
  document.querySelectorAll('.tab-btn').forEach(b => {
    if (b.textContent.toLowerCase().includes(name.substring(0,5))) b.classList.add('active');
  });
  // fix tab active by index
  const names = ['publicacao','aparencia','projetos','config'];
  const idx = names.indexOf(name);
  document.querySelectorAll('.tab-btn')[idx]?.classList.add('active');
}

let dragSrc = null;

function uid() { return Date.now().toString(36) + Math.random().toString(36).slice(2,6); }

function addBlock(type) {
  if (!appData.post) appData.post = { blocks: [] };
  const block = { id: uid(), type };
  if (type === 'text')     block.content = '';
  if (type === 'image')    block.src = '';
  if (type === 'audio')    { block.src = ''; block.label = ''; }
  if (type === 'progress') { block.label = 'Progresso'; block.value = 0; }
  appData.post.blocks.push(block);
  renderBlockList();
  // auto-open last
  const items = document.querySelectorAll('.block-item');
  if (items.length) openBlockEditor(items[items.length-1]);
}

function removeBlock(id) {
  appData.post.blocks = appData.post.blocks.filter(b => b.id !== id);
  renderBlockList();
  refreshPostUI();
}

function renderBlockList() {
  const ul = document.getElementById('block-list');
  ul.innerHTML = '';
  if (!appData.post || !appData.post.blocks) return;
  appData.post.blocks.forEach((block, idx) => {
    const li = document.createElement('li');
    li.className = 'block-item';
    li.draggable = true;
    li.dataset.id = block.id;

    const preview = getBlockPreview(block);
    const typeName = { text:'Texto', image:'Imagem', audio:'Áudio', progress:'Progresso', divider:'Divisor' }[block.type] || block.type;

    li.innerHTML = `
      
        
        ⠿
        ${typeName}
        ${preview}
        
        
          
          
        
        
      
        
      
        ${buildBlockEditorHtml(block)}
        
    `;

    li.querySelector('.block-del').onclick = e => { e.stopPropagation(); removeBlock(block.id); };
    li.querySelector('.block-edit-btn').onclick = e => { e.stopPropagation(); toggleBlockEditor(li); };
    li.querySelector('.block-header').onclick = () => toggleBlockEditor(li);

    // Bind inputs
    bindBlockInputs(li, block);

    // Drag events
    li.addEventListener('dragstart', e => { dragSrc = li; li.classList.add('dragging'); e.dataTransfer.effectAllowed = 'move'; });
    li.addEventListener('dragend', () => { li.classList.remove('dragging'); document.querySelectorAll('.block-item').forEach(i=>i.classList.remove('dragover')); });
    li.addEventListener('dragover', e => { e.preventDefault(); li.classList.add('dragover'); e.dataTransfer.dropEffect = 'move'; });
    li.addEventListener('dragleave', () => li.classList.remove('dragover'));
    li.addEventListener('drop', e => {
      e.preventDefault(); li.classList.remove('dragover');
      if (dragSrc && dragSrc !== li) {
        const srcId = dragSrc.dataset.id;
        const dstId = li.dataset.id;
        const blocks = appData.post.blocks;
        const si = blocks.findIndex(b=>b.id===srcId);
        const di = blocks.findIndex(b=>b.id===dstId);
        if (si > -1 && di > -1) {
          const [moved] = blocks.splice(si, 1);
          blocks.splice(di, 0, moved);
          renderBlockList();
        }
      }
    });

    ul.appendChild(li);
  });
  refreshPostUI();
}

function getBlockPreview(b) {
  if (b.type === 'text') return esc(b.content?.slice(0,50) || '(vazio)');
  if (b.type === 'image') return b.src ? esc(b.src.slice(0,50)+'...') : '(sem URL)';
  if (b.type === 'audio') return b.src ? esc(b.label||b.src.slice(0,40)) : '(sem URL)';
  if (b.type === 'progress') return `${esc(b.label||'Progresso')} — ${b.value||0}%`;
  if (b.type === 'divider') return '———';
  return '';
}

function buildBlockEditorHtml(b) {
  if (b.type === 'text') return `

  if (b.type === 'image') return `

  if (b.type === 'audio') return `
    
        
    
        `;
  if (b.type === 'progress') return `
    
        
    
        
      
        
        
        ${b.value||0}%
      
        
    
        `;
  if (b.type === 'divider') return `
        Linha divisória. Sem opções.
        `;
  return '';
}

function bindBlockInputs(li, block) {
  const ed = li.querySelector('.block-editor');

  const beText  = ed.querySelector('.be-text');
  const beSrc   = ed.querySelector('.be-src');
  const beLabel = ed.querySelector('.be-label');
  const beVal   = ed.querySelector('.be-value');

  if (beText)  beText.oninput  = () => { block.content = beText.value; li.querySelector('.block-preview').textContent = block.content.slice(0,50)||'(vazio)'; };
  if (beSrc)   beSrc.oninput   = () => { block.src = beSrc.value; li.querySelector('.block-preview').textContent = block.src.slice(0,50)||'(sem URL)'; };
  if (beLabel) beLabel.oninput = () => { block.label = beLabel.value; };

  if (beVal) {
    beVal.oninput = () => {
      block.value = parseInt(beVal.value);
      const show = ed.querySelector('.be-val-show');
      const lblShow = ed.querySelector('.be-val-label');
      if (show) show.textContent = block.value + '%';
      if (lblShow) lblShow.textContent = block.value + '%';
      li.querySelector('.block-preview').textContent = `${block.label||'Progresso'} — ${block.value}%`;
    };
  }
}

function toggleBlockEditor(li) {
  const ed = li.querySelector('.block-editor');
  const isOpen = ed.classList.contains('open');
  // close all others
  document.querySelectorAll('.block-editor.open').forEach(e => e.classList.remove('open'));
  if (!isOpen) ed.classList.add('open');
}
function openBlockEditor(li) {
  document.querySelectorAll('.block-editor.open').forEach(e => e.classList.remove('open'));
  li.querySelector('.block-editor').classList.add('open');
}

function loadSettingsUI() {
  const s = appData.settings || {};
  const t = s.backgroundType || 'color';
  document.getElementById('bg-type').value = t;
  if (t === 'color') {
    document.getElementById('bg-color-pick').value = s.backgroundValue || '#000000';
    document.getElementById('bg-color-text').value = s.backgroundValue || '#000000';
  } else {
    document.getElementById('bg-image-url').value = s.backgroundValue || '';
  }
  const bc = s.btnColor || '#aaaaaa';
  document.getElementById('btn-color-pick').value = bc;
  document.getElementById('btn-color-text').value = bc;
  updateBtnPreview(bc);
  toggleBgInput();
  refreshExtraButtonsUI();
}

function collectSettings() {
  const t = document.getElementById('bg-type').value;
  const v = t === 'color'
    ? document.getElementById('bg-color-text').value
    : document.getElementById('bg-image-url').value;
  const btnColor = document.getElementById('btn-color-text').value || '#aaaaaa';
  appData.settings = { backgroundType: t, backgroundValue: v, btnColor };
}

function toggleBgInput() {
  const t = document.getElementById('bg-type').value;
  document.getElementById('bg-color-field').style.display = t === 'color' ? 'block' : 'none';
  document.getElementById('bg-image-field').style.display = t === 'image' ? 'block' : 'none';
}

function syncColorText() {
  document.getElementById('bg-color-text').value = document.getElementById('bg-color-pick').value;
}
function syncColorPick() {
  const v = document.getElementById('bg-color-text').value;
  if (/^#[0-9a-fA-F]{6}$/.test(v)) document.getElementById('bg-color-pick').value = v;
}
function syncBtnColorText() {
  const v = document.getElementById('btn-color-pick').value;
  document.getElementById('btn-color-text').value = v;
  updateBtnPreview(v);
}
function syncBtnColorPick() {
  const v = document.getElementById('btn-color-text').value;
  if (/^#[0-9a-fA-F]{3,6}$/.test(v)) {
    document.getElementById('btn-color-pick').value = v;
    updateBtnPreview(v);
  }
}
function updateBtnPreview(color) {
  const el = document.getElementById('btn-preview-label');
  if (el) { el.style.color = color; el.style.borderColor = color; }
}

function changePassword() {
  const old  = document.getElementById('old-pass').value;
  const nw   = document.getElementById('new-pass').value;
  const conf = document.getElementById('confirm-pass').value;
  if (old !== managerPass) { toast('Password atual incorreta', 'err'); return; }
  if (!nw) { toast('Nova password não pode ser vazia', 'err'); return; }
  if (nw !== conf) { toast('As passwords não coincidem', 'err'); return; }
  managerPass = nw;
  localStorage.setItem('mgr_pass', nw);
  document.getElementById('old-pass').value  = '';
  document.getElementById('new-pass').value  = '';
  document.getElementById('confirm-pass').value = '';
  toast('Password alterada com sucesso!', 'ok');
}

function esc(s) { return String(s||'').replace(/&/g,'&').replace(//g,'>'); }
function escAttr(s) { return String(s||'').replace(/"/g,'"').replace(/'/g,'''); }

let toastTimer;
function toast(msg, type='ok') {
  clearTimeout(toastTimer);
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'show ' + type;
  toastTimer = setTimeout(() => { el.className = ''; }, 2800);
}
